** treemacs
#+begin_src emacs-lisp
  (use-package treemacs
    :defer t
    :bind(:map treemacs-mode-map

	       ("SPC m a" . org-agenda)
	       ("i" . treemacs-previous-line)

	       ("k" . treemacs-next-line)
